CREATE TABLE player(username varchar(20) NOT NULL PRIMARY KEY, password text NOT NULL, ELO int NOT NULL, coins int NOT NULL, salt bytea NOT NULL);

CREATE TABLE card(ID SERIAL NOT NULL PRIMARY KEY, name varchar(255) NOT NULL, element int NOT NULL, damage int NOT NULL);

CREATE TABLE monster(ID SERIAL NOT NULL PRIMARY KEY, MonsterType int NOT NULL, FOREIGN KEY (ID) REFERENCES card(ID));

CREATE TABLE spell(ID SERIAL NOT NULL PRIMARY KEY, FOREIGN KEY (ID) REFERENCES card(ID));
